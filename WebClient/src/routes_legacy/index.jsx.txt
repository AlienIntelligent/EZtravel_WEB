import { createBrowserRouter, Navigate } from "react-router-dom";
import MainLayout from "../layouts/MainLayout";
import AuthLayout from "../layouts/AuthLayout";
import AdminLayout from "../layouts/AdminLayout";
import ProviderLayout from "../layouts/ProviderLayout";
import AILayout from "../layouts/AILayout";

// Auth Guards
import { ProtectedRoute } from "../components/auth/ProtectedRoute";
import { RoleGuard } from "../components/auth/RoleGuard";
import { GuestGuard } from "../components/auth/GuestGuard";

// Pages
import Home from "../modules/home/Home";
import ExploreWorkspace from "../modules/explore/ExploreWorkspace";
import TripPlannerWorkspace from "../modules/trip/TripPlannerWorkspace";
import DesignSystemShowcase from "../modules/design-system/DesignSystemShowcase";
import CommunityWorkspace from "../modules/community/CommunityWorkspace";

// Auth Pages
import Login from "../modules/auth/Login";
import Register from "../modules/auth/Register";
import ForgotPassword from "../modules/auth/ForgotPassword";
import OtpVerification from "../modules/auth/OtpVerification";
import ResetPassword from "../modules/auth/ResetPassword";
import EmailVerificationSuccess from "../modules/auth/EmailVerificationSuccess";

// Admin Pages
import AdminDashboard from "../modules/admin/Dashboard";
import UserManager from "../modules/admin/UserManager";
import PlacesManager from "../modules/admin/PlacesManager";
import ProviderApproval from "../modules/admin/ProviderApproval";
import ServiceModeration from "../modules/admin/ServiceModeration";
import BlogModeration from "../modules/admin/BlogModeration";
import Reports from "../modules/admin/Reports";
import AdminPackagesManager from "../modules/admin/PackageManager";
import AdminPromotionsPreview from "../modules/admin/PromotionsPreview";

// Provider Pages
import ProviderDashboard from "../modules/provider/Dashboard";
import ServicesManager from "../modules/provider/ServicesManager";
import ProviderPackages from "../modules/provider/Packages";
import ProviderCurrentPackage from "../modules/provider/CurrentPackage";
import ProviderPackageHistory from "../modules/provider/PackageHistory";
import ProviderPaymentHistory from "../modules/provider/PaymentHistory";
import ProviderPromotions from "../modules/provider/Promotions";

// AI Pages
import Assistant from "../modules/ai/Assistant";
import AIPlanner from "../modules/ai/Planner";
import History from "../modules/ai/History";

import Forbidden from "../pages/Forbidden";

export const router = createBrowserRouter([
{
  path: "/",
  element: <MainLayout />,
  children: [
  { index: true, element: <Home /> },
  { path: "explore", element: <ExploreWorkspace /> },
  { path: "planner", element: <TripPlannerWorkspace /> },
  { path: "planner/:id", element: <TripPlannerWorkspace /> },
  { path: "community", element: <CommunityWorkspace /> },
  { path: "design-system", element: <DesignSystemShowcase /> }]

},
{
  element:
  <GuestGuard>
        <AuthLayout />
      </GuestGuard>,

  children: [
  { path: "/login", element: <Login /> },
  { path: "/register", element: <Register /> },
  { path: "/forgot-password", element: <ForgotPassword /> },
  { path: "/verify-otp", element: <OtpVerification /> },
  { path: "/reset-password", element: <ResetPassword /> },
  { path: "/verification-success", element: <EmailVerificationSuccess /> }]

},
{
  path: "/admin",
  element:
  <ProtectedRoute>
        <RoleGuard allowedRoles={["ADMIN"]}>
          <AdminLayout />
        </RoleGuard>
      </ProtectedRoute>,

  children: [
  { index: true, element: <AdminDashboard /> },
  { path: "users", element: <UserManager /> },
  { path: "places", element: <PlacesManager /> },
  { path: "providers", element: <ProviderApproval /> },
  { path: "services", element: <ServiceModeration /> },
  { path: "blogs", element: <BlogModeration /> },
  { path: "reports", element: <Reports /> },
  { path: "provider-packages", element: <AdminPackagesManager /> },
  { path: "promotions-preview", element: <AdminPromotionsPreview /> }]

},
{
  path: "/provider",
  element:
  <ProtectedRoute>
        <RoleGuard allowedRoles={["PROVIDER"]}>
          <ProviderLayout />
        </RoleGuard>
      </ProtectedRoute>,

  children: [
  { index: true, element: <Navigate to="dashboard" replace /> },
  { path: "dashboard", element: <ProviderDashboard /> },
  { path: "packages", element: <ProviderPackages /> },
  { path: "current-package", element: <ProviderCurrentPackage /> },
  { path: "package-history", element: <ProviderPackageHistory /> },
  { path: "payment-history", element: <ProviderPaymentHistory /> },
  { path: "promotions", element: <ProviderPromotions /> },
  { path: "services", element: <ServicesManager /> }]

},
{
  path: "/ai",
  element: <AILayout />,
  children: [
  { index: true, element: <Assistant /> },
  { path: "planner", element: <AIPlanner /> },
  { path: "history", element: <History /> }]

},
{
  path: "/unauthorized",
  element: <Forbidden />
},
{
  path: "*",
  element:
  <div className="p-8 text-center">
        <h1 className="text-4xl font-bold">404</h1>
        <p>Không tìm thấy trang</p>
      </div>

}]
);
