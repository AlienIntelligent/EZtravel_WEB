/* eslint-disable no-unused-vars, react-hooks/set-state-in-effect */
import { useAppSelector } from '../../store/hooks';
import { useAnalyzeBudgetMutation } from '../../api/aiApi';

function AIBudgetPanel() {
  const trip = useAppSelector((state) => state.trip.activeTrip);
  const [optimizeBudget, { data, isLoading, isUninitialized }] = useAnalyzeBudgetMutation();

  if (!trip) return null;

  const handleOptimize = async () => {
    try {
      await optimizeBudget({ tripId: trip.id }).unwrap();
    } catch (error) {
      alert('Không thể tối ưu ngân sách lúc này.');
    }
  };

  return (
    <div className="card shadow-sm border-0 h-100" style={{ borderRadius: '10px' }}>
        <div className="card-header bg-success text-white p-3 d-flex justify-content-between align-items-center">
            <h6 className="m-0 fw-bold"><i className="fas fa-piggy-bank me-2"></i> AI Tối ưu Ngân sách</h6>
        </div>
        <div className="card-body p-3 overflow-auto">
            <p className="small text-muted mb-3">AI sẽ phân tích các dịch vụ trong lịch trình và đề xuất các dịch vụ thay thế từ hệ thống để tiết kiệm chi phí mà vẫn đảm bảo trải nghiệm.</p>
            
            {isUninitialized &&
        <button className="btn btn-outline-success w-100" onClick={handleOptimize}>
                    <i className="fas fa-magic me-2"></i> Phân tích Ngân sách
                </button>
        }

            {isLoading &&
        <div className="text-center text-success py-4">
                    <i className="fas fa-circle-notch fa-spin fa-2x mb-2"></i>
                    <p className="mb-0 small">Đang quét cơ sở dữ liệu dịch vụ...</p>
                </div>
        }

            {data &&
        <div>
                    <div className="alert alert-success border-0 py-2 px-3 mb-3 d-flex justify-content-between align-items-center">
                        <span className="fw-bold">Tiết kiệm dự kiến:</span>
                        <span className="fs-5 fw-bold text-success">{data.suggestedSavings.toLocaleString()} đ</span>
                    </div>

                    {data.optimizations.length === 0 ?
          <div className="text-center text-muted small py-3">Lịch trình của bạn đã rất tối ưu về chi phí!</div> :

          <div className="d-flex flex-column gap-3">
                            {data.optimizations.map((opt, idx) =>
            <div key={idx} className="border rounded p-2 bg-light">
                                    <div className="d-flex justify-content-between align-items-start mb-2">
                                        <span className="badge bg-danger text-decoration-line-through">{opt.originalCost.toLocaleString()} đ</span>
                                        <i className="fas fa-arrow-right text-muted"></i>
                                        <span className="badge bg-success">{opt.suggestedService.price.toLocaleString()} đ</span>
                                    </div>
                                    <div className="small fw-bold mb-1">Thay thế: {opt.originalServiceName}</div>
                                    <div className="small text-primary mb-1">Bằng: {opt.suggestedService.name}</div>
                                    <div className="text-muted" style={{ fontSize: '11px' }}>
                                        <i className="fas fa-info-circle me-1"></i> {opt.reason}
                                    </div>
                                </div>
            )}
                        </div>
          }
                    <button className="btn btn-sm btn-outline-secondary w-100 mt-3" onClick={handleOptimize}>
                        <i className="fas fa-sync-alt me-2"></i> Phân tích lại
                    </button>
                </div>
        }
        </div>
    </div>);

}

export default AIBudgetPanel;
