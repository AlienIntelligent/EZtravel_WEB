/* eslint-disable no-unused-vars, react-hooks/set-state-in-effect */
import { useAppDispatch, useAppSelector } from '../../store/hooks';
import { setTimelineDays } from '../../store/tripSlice';
import { useUpdateSequenceMutation } from '../../api/tripApi';
import { useOptimizeRouteMutation } from '../../api/aiApi';

function AIRoutePanel() {
  const dispatch = useAppDispatch();
  const trip = useAppSelector((state) => state.trip.activeTrip);
  const timelineDays = useAppSelector((state) => state.trip.timelineDays);
  const [optimizeRoute, { data, isLoading, isUninitialized }] = useOptimizeRouteMutation();
  const [updateSequenceApi] = useUpdateSequenceMutation();

  if (!trip) return null;

  const handleOptimize = async () => {
    try {
      await optimizeRoute({ tripId: trip.id }).unwrap();
    } catch (error) {
      alert('Không thể tối ưu lộ trình lúc này.');
    }
  };

  const handleApply = async (dayId, optimizedOrder) => {
    if (window.confirm('Bạn có chắc chắn muốn thay đổi thứ tự các địa điểm trong ngày này theo đề xuất của AI?')) {
      const dayIndex = timelineDays.findIndex((d) => d.id === dayId);
      if (dayIndex === -1) return;
      const day = timelineDays[dayIndex];

      // Reorder nodes
      const newNodes = [...day.nodes].sort((a, b) => {
        const idA = a.type === 'TIMELINE_ITEM' ? a.item.id : a.id;
        const idB = b.type === 'TIMELINE_ITEM' ? b.item.id : b.id;
        return optimizedOrder.indexOf(idA) - optimizedOrder.indexOf(idB);
      });
      newNodes.forEach((node, idx) => node.sequence = idx);

      const newDays = [...timelineDays];
      newDays[dayIndex] = { ...day, nodes: newNodes };
      dispatch(setTimelineDays(newDays));

      try {
        const updates = newNodes.map((node) => {
          const id = node.type === 'TIMELINE_ITEM' ? node.item.id : node.id;
          const type = node.type === 'TIMELINE_ITEM' && 'placeId' in node.item ? 'PLACE' : 'SERVICE';
          return {
            id: id,
            type: type,
            sequence: node.sequence,
            dayId: day.id
          };
        });
        await updateSequenceApi({ tripId: trip.id, updates }).unwrap();
        alert('Đã áp dụng lộ trình tối ưu!');
      } catch (err) {
        alert('Có lỗi khi lưu lên server.');
      }
    }
  };

  return (
    <div className="card shadow-sm border-0 h-100" style={{ borderRadius: '10px' }}>
        <div className="card-header bg-primary text-white p-3 d-flex justify-content-between align-items-center">
            <h6 className="m-0 fw-bold"><i className="fas fa-route me-2"></i> AI Tối ưu Lộ trình</h6>
        </div>
        <div className="card-body p-3 overflow-auto">
            <p className="small text-muted mb-3">AI sẽ phân tích khoảng cách và tình trạng giao thông để đề xuất thứ tự di chuyển hợp lý nhất giữa các địa điểm.</p>
            
            {isUninitialized &&
        <button className="btn btn-outline-primary w-100" onClick={handleOptimize}>
                    <i className="fas fa-magic me-2"></i> Phân tích Lộ trình
                </button>
        }

            {isLoading &&
        <div className="text-center text-primary py-4">
                    <i className="fas fa-circle-notch fa-spin fa-2x mb-2"></i>
                    <p className="mb-0 small">Đang tính toán ma trận khoảng cách...</p>
                </div>
        }

            {data &&
        <div>
                    {data.suggestions.length === 0 ?
          <div className="text-center text-muted small py-3">Thứ tự các địa điểm của bạn đã rất tối ưu!</div> :

          <div className="d-flex flex-column gap-3">
                            {data.suggestions.map((sugg, idx) => {
              const day = timelineDays.find((d) => d.id === sugg.dayId);
              if (!day) return null;

              return (
                <div key={idx} className="border rounded p-3 bg-light">
                                        <div className="fw-bold text-dark mb-1">
                                            <i className="fas fa-calendar-day text-primary me-2"></i> Ngày {new Date(day.date).toLocaleDateString('vi-VN')}
                                        </div>
                                        <div className="small text-success fw-bold mb-2">
                                            <i className="fas fa-clock me-1"></i> Tiết kiệm: ~{sugg.estimatedTimeSavedMinutes} phút di chuyển
                                        </div>
                                        <p className="text-muted" style={{ fontSize: '12px' }}>
                                            {sugg.reason}
                                        </p>
                                        <button
                    className="btn btn-sm btn-primary w-100 fw-bold"
                    onClick={() => handleApply(sugg.dayId, sugg.optimizedOrder)}>
                    
                                            Áp dụng thay đổi
                                        </button>
                                    </div>);

            })}
                        </div>
          }
                    <button className="btn btn-sm btn-outline-secondary w-100 mt-3" onClick={handleOptimize}>
                        <i className="fas fa-sync-alt me-2"></i> Phân tích lại
                    </button>
                </div>
        }
        </div>
    </div>);

}

export default AIRoutePanel;
