import React, { useEffect, useMemo, useState } from "react";
import { Link } from "react-router-dom";
import { placeApi } from "../../api";

const Rating = ({ value, reviews }) => (
  <p className="rate">
    {[0, 1, 2, 3, 4].map((index) => (
      <i
        key={index}
        className={index < Math.round(value) ? "icon-star" : "icon-star-o"}
      ></i>
    ))}
    <span>{reviews} đánh giá</span>
  </p>
);

function Places() {
  const [places, setPlaces] = useState([]);
  const [loading, setLoading] = useState(true);

  const [keyword, setKeyword] = useState("");
  const [budget, setBudget] = useState(10000000);

  const [currentPage, setCurrentPage] = useState(1);

  const placesPerPage = 6;

  useEffect(() => {
    const fetchPlaces = async () => {
      try {
        const response = await placeApi.search({ pageSize: 100 });
        const data = response.data;
        setPlaces(data.items || data);
      } catch (error) {
        console.error(error);
      } finally {
        setLoading(false);
      }
    };

    fetchPlaces();
  }, []);

  const filteredPlaces = useMemo(() => {
    return places.filter((place) => {
      const matchKeyword =
        place.tenDiaDiem?.toLowerCase().includes(keyword.toLowerCase()) ||
        place.tenTinhThanh?.toLowerCase().includes(keyword.toLowerCase());

      return matchKeyword;
    });
  }, [places, keyword]);

  const totalPages = Math.ceil(filteredPlaces.length / placesPerPage);

  const startIndex = (currentPage - 1) * placesPerPage;

  const currentPlaces = filteredPlaces.slice(
    startIndex,
    startIndex + placesPerPage,
  );

  const addToPlanner = (place) => {
    const saved =
      JSON.parse(localStorage.getItem("eztravel-selected-places")) || [];

    const exists = saved.find((item) => item.id === place.id);

    if (exists) {
      alert("Địa điểm đã có trong planner");
      return;
    }

    const newPlace = {
      id: place.id,
      name: place.tenDiaDiem,
      location: place.tinhThanh,
      image: `https://picsum.photos/600/400?random=${place.id}`,
      type: "place",
    };

    const updated = [...saved, newPlace];

    localStorage.setItem("eztravel-selected-places", JSON.stringify(updated));

    alert("Đã thêm vào planner");
  };

  return (
    <>
      {/* HERO */}
      <div
        className="hero-wrap js-fullheight"
        style={{
          backgroundImage: 'url("/images/bg_3.jpg")',
        }}
      >
        <div className="overlay"></div>

        <div className="container">
          <div
            className="row no-gutters slider-text js-fullheight align-items-center justify-content-center"
            data-scrollax-parent="true"
          >
            <div className="col-md-9 ftco-animate text-center">
              <p className="breadcrumbs">
                <span className="mr-2">
                  <Link to="/">Trang chủ</Link>
                </span>

                <span>Điểm đến</span>
              </p>

              <h1 className="mb-3 bread">Điểm đến</h1>

              <p className="caps">Khám phá Việt Nam theo cách riêng của bạn.</p>
            </div>
          </div>
        </div>
      </div>

      {/* CONTENT */}
      <section className="ftco-section ftco-degree-bg">
        <div className="container">
          <div className="row">
            {/* SIDEBAR */}
            <div className="col-lg-3 sidebar ftco-animate">
              <div className="sidebar-wrap bg-light ftco-animate">
                <h3 className="heading mb-4">Tìm địa điểm</h3>

                <div className="fields">
                  <div className="form-group">
                    <input
                      type="text"
                      className="form-control"
                      placeholder="Bạn muốn đi đâu?"
                      value={keyword}
                      onChange={(e) => {
                        setKeyword(e.target.value);
                        setCurrentPage(1);
                      }}
                    />
                  </div>

                  <div className="form-group">
                    <label>Ngân sách tối đa</label>

                    <input
                      type="range"
                      className="form-control"
                      min="1000000"
                      max="10000000"
                      step="500000"
                      value={budget}
                      onChange={(e) => setBudget(Number(e.target.value))}
                    />

                    <small>{budget.toLocaleString("vi-VN")}đ</small>
                  </div>

                  <div className="form-group">
                    <button className="btn btn-primary py-3 px-5" type="button">
                      {filteredPlaces.length} kết quả
                    </button>
                  </div>
                </div>
              </div>

              {/* QUICK ACTION */}
              <div className="sidebar-wrap bg-light ftco-animate">
                <h3 className="heading mb-4">Tạo nhanh</h3>

                <button className="btn btn-outline-primary btn-block mb-3">
                  + Tạo lịch trình mới
                </button>

                <button className="btn btn-outline-secondary btn-block">
                  📌 Lịch trình đã lưu
                </button>
              </div>
            </div>

            {/* RIGHT CONTENT */}
            <div className="col-lg-9">
              <div className="row">
                {loading && (
                  <div className="col-12">
                    <h3>Đang tải...</h3>
                  </div>
                )}

                {!loading &&
                  currentPlaces.map((place) => (
                    <div className="col-md-4" key={place.id}>
                      <div className="destination">
                        <Link
                          to={`/places/${place.id}`}
                          className="img img-2 d-flex justify-content-center align-items-center"
                          style={{
                            backgroundImage: `url(https://picsum.photos/600/400?random=${place.id})`,
                            height: "250px",
                            backgroundSize: "cover",
                            backgroundPosition: "center",
                          }}
                        >
                          <div className="icon d-flex justify-content-center align-items-center">
                            <span className="icon-search2"></span>
                          </div>
                        </Link>

                        <div className="text p-3">
                          <div className="d-flex">
                            <div className="one">
                              <h3>
                                <Link to={`/places/${place.id}`}>
                                  {place.tenDiaDiem}
                                </Link>
                              </h3>

                              <p className="rate">
                                <span>{place.tenTinhThanh}</span>
                              </p>
                            </div>

                            <div className="two">
                              <span className="price">HOT</span>
                            </div>
                          </div>

                          <p>{place.moTa}</p>

                          <p className="days">
                            <span>{place.diaChi}</span>
                          </p>

                          <hr />

                          <p className="bottom-area d-flex justify-content-between align-items-center">
                            <span>
                              <i className="icon-map-o"></i> {place.tenTinhThanh}
                            </span>

                            <div>
                              <button
                                className="btn btn-sm btn-outline-primary mr-2"
                                onClick={() => addToPlanner(place)}
                              >
                                + Thêm
                              </button>

                              <Link
                                to={`/places/${place.id}`}
                                className="btn btn-sm btn-primary"
                              >
                                Xem
                              </Link>
                            </div>
                          </p>
                        </div>
                      </div>
                    </div>
                  ))}
              </div>

              {filteredPlaces.length === 0 && (
                <div className="alert alert-light border">
                  Không có địa điểm phù hợp.
                </div>
              )}

              {/* PAGINATION */}
              <div className="row mt-5">
                <div className="col text-center">
                  <div className="block-27">
                    <ul>
                      <li>
                        <a
                          href="#!"
                          onClick={(e) => {
                            e.preventDefault();

                            setCurrentPage((prev) => Math.max(prev - 1, 1));
                          }}
                        >
                          &lt;
                        </a>
                      </li>

                      {Array.from({ length: totalPages }, (_, index) => (
                        <li
                          key={index}
                          className={currentPage === index + 1 ? "active" : ""}
                        >
                          <a
                            href="#!"
                            onClick={(e) => {
                              e.preventDefault();
                              setCurrentPage(index + 1);
                            }}
                          >
                            {index + 1}
                          </a>
                        </li>
                      ))}

                      <li>
                        <a
                          href="#!"
                          onClick={(e) => {
                            e.preventDefault();

                            setCurrentPage((prev) =>
                              Math.min(prev + 1, totalPages),
                            );
                          }}
                        >
                          &gt;
                        </a>
                      </li>
                    </ul>
                  </div>
                </div>
              </div>
            </div>
          </div>
        </div>
      </section>
    </>
  );
}

export default Places;
