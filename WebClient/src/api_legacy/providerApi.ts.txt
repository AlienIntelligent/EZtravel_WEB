import { baseApi } from './baseApi';
import { Service, Review } from '../shared/types';
import { ProviderState } from '../store/providerSlice';

export interface ProviderDashboardKPIs {
    totalServices: number;
    activeServices: number;
    totalReviews: number;
    averageRating: number;
}

export interface PaginatedResponse<T> {
    data: T[];
    total: number;
    page: number;
    pageSize: number;
    totalPages: number;
}

export const providerApi = baseApi.injectEndpoints({
    endpoints: (builder) => ({
        getProviderDashboard: builder.query<ProviderDashboardKPIs, number>({
            query: (userId) => ({ url: `/providers/user/${userId}/dashboard`, method: 'GET' }),
            providesTags: ['Provider'],
        }),
        getProviderServices: builder.query<PaginatedResponse<Service>, Partial<ProviderState['servicesFilter']> & { providerId: number }>({
            queryFn: async (filters, _queryApi, _extraOptions, fetchWithBQ) => {
                const params = new URLSearchParams();
                if (filters.keyword) params.append('keyword', filters.keyword);
                params.append('maNhaCungCap', filters.providerId.toString());
                params.append('page', (filters.page || 1).toString());
                params.append('pageSize', '10');

                const queryStr = params.toString();
                let results: any[] = [];
                
                try {
                    const fetchCategory = async (cat: string) => {
                        const result = await fetchWithBQ({ url: `/places/${cat}/search?${queryStr}`, method: 'GET' });
                        const data = result.data ? (result.data as any[]) : [];
                        return data.map((item: any) => {
                            let name = '';
                            let category = 'ACCOMMODATION';
                            if (cat === 'hotels') { name = item.tenKhachSan; category = 'ACCOMMODATION'; }
                            if (cat === 'restaurants') { name = item.tenNhaHang; category = 'FOOD'; }
                            if (cat === 'activities') { name = item.tenHoatDong; category = 'ACTIVITY'; }
                            if (cat === 'vehicles') { name = item.tenPhuongTien; category = 'TRANSPORT'; }

                            return {
                                id: item.id?.toString() || '',
                                providerId: (item.maNhaCungCap || 0).toString(),
                                placeId: (item.maDiaDiem || 0).toString(),
                                name: name,
                                description: item.moTa || '',
                                type: category,
                                price: item.giaTu || item.giaTrungBinh || item.price || 0,
                                referencePrice: item.giaDen,
                                images: item.anhDaiDien ? [item.anhDaiDien] : [],
                                averageRating: item.ratingAvg || 0,
                                totalReviews: item.totalReviews || 0,
                                status: item.trangThai || 'ACTIVE',
                                createdAt: item.ngayTao || new Date().toISOString(),
                                updatedAt: item.ngayCapNhat || new Date().toISOString(),
                            };
                        });
                    };

                    const [hotels, restaurants, activities, vehicles] = await Promise.all([
                        fetchCategory('hotels'),
                        fetchCategory('restaurants'),
                        fetchCategory('activities'),
                        fetchCategory('vehicles')
                    ]);
                    results = [...hotels, ...restaurants, ...activities, ...vehicles];
                    
                    // Simple pagination slice since we merged
                    results.sort((a, b) => (b.id || 0) - (a.id || 0)); // Sort by newest (id)
                    const total = results.length;
                    const page = filters.page || 1;
                    const pageSize = 10;
                    const paginatedResults = results.slice((page - 1) * pageSize, page * pageSize);
                    
                    return { data: { data: paginatedResults, total, page, pageSize, totalPages: Math.ceil(total / pageSize) } };
                } catch (error: any) {
                    return { error: error.message };
                }
            },
            providesTags: (result) => 
                result
                    ? [
                        ...result.data.map(({ id }) => ({ type: 'Service' as const, id })),
                        { type: 'Service', id: 'LIST' }
                    ]
                    : [{ type: 'Service', id: 'LIST' }],
        }),
        createService: builder.mutation<Service, Partial<Service>>({
            query: (newService) => ({
                url: '/provider/services',
                method: 'POST',
                body: newService,
            }),
            invalidatesTags: [{ type: 'Service', id: 'LIST' }, 'Provider'],
        }),
        updateService: builder.mutation<Service, { id: string; data: Partial<Service> }>({
            query: ({ id, data }) => ({
                url: `/provider/services/${id}`,
                method: 'PUT',
                body: data,
            }),
            invalidatesTags: (result, error, { id }) => [{ type: 'Service', id }, 'Provider'],
        }),
        deleteService: builder.mutation<void, string>({
            // Soft delete by setting status to INACTIVE
            query: (id) => ({
                url: `/provider/services/${id}`,
                method: 'DELETE', // Assuming backend treats DELETE as soft-delete if it maps to INACTIVE
            }),
            invalidatesTags: (result, error, id) => [{ type: 'Service', id }, { type: 'Service', id: 'LIST' }, 'Provider'],
        }),
        getProviderReviews: builder.query<PaginatedResponse<Review>, { page: number }>({
            query: ({ page }) => ({ url: `/provider/reviews?page=${page}&pageSize=10`, method: 'GET' }),
            providesTags: ['Review'],
        }),
        // Analytics
        getProviderAnalytics: builder.query<any, void>({
            query: () => ({ url: '/provider/analytics', method: 'GET' }),
            providesTags: ['Provider'],
        })
    }),
    overrideExisting: false,
});

export const { 
    useGetProviderDashboardQuery, 
    useGetProviderServicesQuery,
    useCreateServiceMutation,
    useUpdateServiceMutation,
    useDeleteServiceMutation,
    useGetProviderReviewsQuery,
    useGetProviderAnalyticsQuery
} = providerApi;
